        // Update countdown every second

        function updateCountdown() {

            const now = new Date();

            const tomorrow = new Date(now);

            tomorrow.setDate(tomorrow.getDate() + 1);

            tomorrow.setHours(4, 0, 0, 0); // Tomorrow's Fajr at 4:00

            

            const timeDiff = tomorrow - now;

            

            if (timeDiff > 0) {

                const hours = Math.floor(timeDiff / (1000 * 60 * 60));

                const minutes = Math.floor((timeDiff % (1000 * 60 * 60)) / (1000 * 60));

                const seconds = Math.floor((timeDiff % (1000 * 60)) / 1000);

                

                document.getElementById('countdown').textContent = 

                    `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;

            }

            

            // Update prayer highlighting

            updatePrayerHighlighting();

        }


        function updatePrayerHighlighting() {

            const now = new Date();

            const todayTimes = {

                fajr: { jamah: "4:00", begin: "2:55" },

                zuhr: { jamah: "1:30", begin: "1:10" },

                asr: { jamah: "7:30", begin: "6:40" },

                maghrib: { jamah: "9:27", begin: "9:22" },

                isha: { jamah: "11:00", begin: "10:38" }

            };


            // Parse prayer times - using Jamah for start and next prayer's BEGIN for end

            const prayers = [

                { 

                    name: 'Fajr', 

                    startTime: parseTime(todayTimes.fajr.jamah),

                    endTime: parseTime(todayTimes.zuhr.begin) // Until Zuhr begins

                },

                { 

                    name: 'Zuhr', 

                    startTime: parseTime(todayTimes.zuhr.jamah),

                    endTime: parseTime(todayTimes.asr.begin) // Until Asr begins

                },

                { 

                    name: 'Asr', 

                    startTime: parseTime(todayTimes.asr.jamah),

                    endTime: parseTime(todayTimes.maghrib.begin) // Until Maghrib begins

                },

                { 

                    name: 'Maghrib', 

                    startTime: parseTime(todayTimes.maghrib.jamah),

                    endTime: parseTime(todayTimes.isha.begin) // Until Isha begins

                },

                { 

                    name: 'Isha', 

                    startTime: parseTime(todayTimes.isha.jamah),

                    endTime: (() => {

                        // Until tomorrow's Fajr begins

                        const tomorrowFajr = parseTime("2:55");

                        tomorrowFajr.setDate(tomorrowFajr.getDate() + 1);

                        return tomorrowFajr;

                    })()

                }

            ];


            // Find current and next prayer

            let currentPrayer = null;

            let nextPrayer = null;


            // Check if we're currently in a prayer window

            for (let prayer of prayers) {

                if (now >= prayer.startTime && now < prayer.endTime) {

                    currentPrayer = prayer.name;

                    break;

                }

            }


            // Find next prayer

            for (let prayer of prayers) {

                if (now < prayer.startTime) {

                    nextPrayer = prayer.name;

                    break;

                }

            }


            // If no next prayer today, next is tomorrow's Fajr

            if (!nextPrayer) {

                nextPrayer = 'Fajr';

            }


            // Update prayer name in next prayer section

            document.getElementById('nextPrayerName').textContent = nextPrayer;


            // Remove existing classes

            document.querySelectorAll('.prayer-row').forEach(row => {

                row.classList.remove('current', 'next');

                const statusElement = row.querySelector('.prayer-details p');

                statusElement.textContent = '';

            });


            // Add current prayer highlighting

            if (currentPrayer) {

                const currentRow = document.querySelector(`[data-prayer="${currentPrayer.toLowerCase()}"]`);

                if (currentRow) {

                    currentRow.classList.add('current');

                    const statusElement = currentRow.querySelector('.prayer-details p');

                    statusElement.textContent = 'Current Prayer';

                }

            }


            // Add next prayer highlighting

            const nextRow = document.querySelector(`[data-prayer="${nextPrayer.toLowerCase()}"]`);

            if (nextRow) {

                nextRow.classList.add('next');

                const statusElement = nextRow.querySelector('.prayer-details p');

                statusElement.textContent = 'Next Prayer';

            }

        }


        function parseTime(timeStr) {

            const [hours, minutes] = timeStr.split(':').map(Number);

            const date = new Date();

            date.setHours(hours, minutes, 0, 0);

            return date;

        }


        // Start countdown

        setInterval(updateCountdown, 1000);

        updateCountdown(); // Run immediately

        // Update countdown every second

        function updateCountdown() {

            const now = new Date();

            const tomorrow = new Date(now);

            tomorrow.setDate(tomorrow.getDate() + 1);

            tomorrow.setHours(4, 0, 0, 0); // Tomorrow's Fajr at 4:00

            

            const timeDiff = tomorrow - now;

            

            if (timeDiff > 0) {

                const hours = Math.floor(timeDiff / (1000 * 60 * 60));

                const minutes = Math.floor((timeDiff % (1000 * 60 * 60)) / (1000 * 60));

                const seconds = Math.floor((timeDiff % (1000 * 60)) / 1000);

                

                document.getElementById('countdown').textContent = 

                    `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;

            }

        }


        // Start countdown

        setInterval(updateCountdown, 1000);

        updateCountdown(); // Run immediately

    </script>
